90int (transform_color)(uint32_t color, uint32_t *new_color) {
91 if (new_color == NULL) return 1;
92 uint8_t red, green, blue;
93 switch (mode_info.BitsPerPixel) {
94 // 1024x768
95 case 8: {
96 *new_color = color & 0x0000000F;
97 return 0;
98 }
99 // 640x480
100 case 15: {
101 red = (color & 0x00007C00) >> 8;
102 green = (color & 0x000003E0) >> 4;
103 blue = (color & 0x0000001F);
104 *new_color = red | green | blue;
105 return 0;
106 }
107 // 800x600
108 case 24: {
109 red = (color & 0x00FF0000) >> 16;
110 green = (color & 0x0000FF00) >> 8;
111 blue = (color & 0x000000FF);
112 *new_color = red | green | blue;
113 return 0;
114 }
115 // 1280x1024
116 case 16 : {
117 red = (color & 0x0000F800) >> 8;
118 green = (color & 0x000007E0) >> 3;
119 blue = (color & 0x0000001F);
120 *new_color = red | green | blue;
121 return 0;
122 }
123 // 1152x864
124 case 32: {
125 red = (color & 0x00FF0000) >> 16;
126 green = (color & 0x0000FF00) >> 8;
127 blue = (color & 0x000000FF);
128 *new_color = red | green | blue;
129 return 0;
130 }
131 default:
132 return 1;
133 }
134 return 1;
135}
