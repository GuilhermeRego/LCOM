436int char_to_index(char c) {
437 if (c >= 'A' && c <= 'Z') {
438 is_alpha = true;
439 return c - 'A';
440 }
441 else if (c >= 'a' && c <= 'z') {
442 is_alpha = true;
443 return c - 'a';
444 }
445 else if (c >= '0' && c <= '9') {
446 is_number = true;
447 return c - '0';
448 }
449 else if (c == '/') {
450 is_special = true;
451 return 0;
